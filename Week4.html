sum_x=[]
sum_y=[]
for j in range(0,10000):
    x=np.zeros([1,1])
    y=np.zeros([1,1])
    #generate random uniform points
    amplitude = np.random.uniform(0, 10, 100)
    phase= np.random.uniform(0,np.pi, 100)
    for i in range(0,amplitude.size):
        # Random Phasor Sum in phasor space
        x = x + amplitude[i] * np.cos(phase[i])  #x axis--cos(phase)
        y = y + amplitude[i] * np.sin(phase[i])  #y axis--jsin(phase)
    sum_x.append(x)
    sum_y.append(y)
plt.scatter(sum_x,sum_y,color='black',s=2)
plt.show()

The result of the sum of a large number ( N→ ∞) of random phasors follows Normal distribution
