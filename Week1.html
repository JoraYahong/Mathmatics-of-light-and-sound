# Draw waves using the center of the wave and the wavelength
def drawWave(waveCenter,wavelength):
    # set theta as 1/2 pi for half circle
    theta = np.linspace( 0 , np.pi , 100 )
    # L = radius * α ---- Using Math formula to draw Arc
    for i in range(0,waveCenter.shape[1]):
        for j in wavelength:
            x = j * np.cos( theta )+ waveCenter[0][i]
            y = j * np.sin( theta )+ waveCenter[1][i]
            plt.plot(x,y,color='#97C7DE')
